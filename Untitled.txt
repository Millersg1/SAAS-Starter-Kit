<script>
document.getElementById("year").textContent = new Date().getFullYear();

function setVerse(text, ref){
  document.getElementById("verse").textContent = text;
  document.getElementById("ref").textContent = "— " + ref;
}

function loadVerse(){
  fetch("https://beta.ourmanna.com/api/v1/get?format=json&order=daily")
    .then(r => r.json())
    .then(d => {
      const text = d.verse.details.text.replace(/<[^>]+>/g,'');
      const ref  = d.verse.details.reference;
      setVerse(text, ref);
    })
    .catch(() => {
      setVerse("Commit to the LORD whatever you do, and he will establish your plans.", "Proverbs 16:3");
    });
}

function refreshVerse(){ loadVerse(); }

function copyVerse(){
  const text = document.getElementById("verse").textContent + " " + document.getElementById("ref").textContent;
  if (navigator.clipboard && navigator.clipboard.writeText){
    navigator.clipboard.writeText(text)
      .then(()=>alert("Verse copied!"))
      .catch(()=>alert("Copy failed — please copy manually."));
  } else {
    alert("Copy not supported in this browser — please copy manually.");
  }
}

loadVerse();
